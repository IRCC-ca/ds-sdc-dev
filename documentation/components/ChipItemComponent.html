import { Component, Input, Output, EventEmitter } from '@angular/core';

@Component({
  selector: 'lib-chip-item',
  templateUrl: './chip-item.component.html'
})
export class ChipItemComponent {
  @Input() chipContent?: string;
  @Output() iconClickEvent = new EventEmitter();

  constructor() {}

  onIconClick() {
    this.iconClickEvent.emit();
  }

  onEnterKeyPress() {
    this.iconClickEvent.emit();
  }
}

<div class="chip-item" tabindex="0" (keydown.enter)="onEnterKeyPress()">
  <span class="chip-text">{{ chipContent || '' }}</span>
  <span class="icon-container" (click)="onIconClick()">
    <i class="fa-solid fa-circle-xmark clickable-icon"></i>
  </span>
</div>
